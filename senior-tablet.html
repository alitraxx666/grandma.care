function updateClock() { const now = new Date(); const options = { weekday: "long", month: "long", day: "numeric" }; const dateString = now.toLocaleDateString("en-US", options); const timeString = now.toLocaleTimeString("en-US", { hour: "numeric", minute: "2-digit" }); document.getElementById("clockDisplay").innerHTML = `${dateString}
${timeString}`; } updateClock(); setInterval(updateClock, 1000);
